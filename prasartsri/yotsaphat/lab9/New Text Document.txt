package prasartsri.yotsaphat.lab9;

import java.awt.image.BufferedImage;
import java.io.File;
import java.io.IOException;
import javax.imageio.ImageIO;
import javax.swing.JPanel;
import java.awt.*;
import javax.swing.*;
import javax.swing.SwingUtilities;

public class ReadImage extends JPanel {
    public BufferedImage img
    public String fileName = "prasartsri/yotsaphat/lab9/images/S25-ultra.jpg";

    public void paintComponent(String picturename) {
        myImage = new ReadWindowImage(picturename);
        myImage.setPreferredSize(new Dimension(600, 450));

    }

    public ReadImage() {
        try {
            img = ImageIO.read(new File(fileName)); // Read the image file
        } catch (IOException e) { // Catch the exception with correct spelling
            e.printStackTrace(System.err); // Print the error
        }
    }
    
}
